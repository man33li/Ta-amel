**Mock useNotes hook** - mock at module level to control different states:

```typescript
import { describe, it, expect, vi, beforeEach } from 'vitest'
import { render, screen, waitFor } from '@testing-library/react'
import userEvent from '@testing-library/user-event'
import HomePage from '@/app/(main)/page'
import type { Card } from '@/types'

// Mock useNotes hook
const mockCreateNote = vi.fn()
vi.mock('@/lib/hooks/useNotes', () => ({
  useNotes: vi.fn()
}))

// Import after mock
import { useNotes } from '@/lib/hooks/useNotes'
const mockedUseNotes = vi.mocked(useNotes)

// Mock router
const mockPush = vi.fn()
vi.mock('next/navigation', () => ({
  useRouter: () => ({ push: mockPush })
}))
```

**Test cases:**

1. **Loading state:**
   - Shows loading skeleton when loading=true
   
2. **Empty state:**
   - Shows "No notes yet" when notes=[]
   - Shows "Create First Note" button
   
3. **Notes list:**
   - Renders NoteCard for each note
   - Shows note titles

4. **Create note:**
   - Clicking create button calls createNote
   - Navigates to new note after creation

5. **Error state:**
   - Shows error message when error exists

Pattern:
```typescript
describe('HomePage', () => {
  beforeEach(() => {
    vi.clearAllMocks()
  })

  it('shows loading skeleton when loading', () => {
    mockedUseNotes.mockReturnValue({
      notes: [],
      loading: true,
      error: null,
      createNote: mockCreateNote,
      updateNote: vi.fn(),
      deleteNote: vi.fn(),
      refetch: vi.fn()
    })
    
    render(<HomePage />)
    expect(screen.getByTestId('loading-skeleton')).toBeInTheDocument()
  })

  it('shows empty state when no notes', () => {
    mockedUseNotes.mockReturnValue({
      notes: [],
      loading: false,
      error: null,
      createNote: mockCreateNote,
      updateNote: vi.fn(),
      deleteNote: vi.fn(),
      refetch: vi.fn()
    })
    
    render(<HomePage />)
    expect(screen.getByText(/no notes yet/i)).toBeInTheDocument()
  })

  it('creates note and navigates on button click', async () => {
    const user = userEvent.setup()
    const newNote = { id: 'new-note-id', title: 'Untitled' } as Card
    mockCreateNote.mockResolvedValue(newNote)
    
    mockedUseNotes.mockReturnValue({
      notes: [],
      loading: false,
      error: null,
      createNote: mockCreateNote,
      updateNote: vi.fn(),
      deleteNote: vi.fn(),
      refetch: vi.fn()
    })
    
    render(<HomePage />)
    await user.click(screen.getByRole('button', { name: /create first note/i }))
    
    expect(mockCreateNote).toHaveBeenCalledWith('Untitled')
    await waitFor(() => {
      expect(mockPush).toHaveBeenCalledWith('/notes/new-note-id')
    })
  })
})
```
